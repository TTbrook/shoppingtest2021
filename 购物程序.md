```python

供选购的产品=[                                   #这里用到了列表 和 元组
    ('苹果手机',9299),
    ('苹果电脑', 11999),
    ('表', 6800),
    ('人工智能初步编程教程', 100),
    ('一杯咖啡', 15),
    ('笔', 5),
    ('笔记本', 10),
]
总价=0
购物清单=[]
print('现有如下商品，单价人民币（元）：')
for 索引,项目 in enumerate(供选购的产品): #enumerate()函数用于将一个可遍历的数据对象(如列表、元组或字符串)组合为一个"索引"序列，同时列出数据和数据下标
    print(索引,项目)
print("要买的东西输入前面的序号，如果不需要买东西，或者停止购买，请输入q")
while 1:
    print()
    选择=input("请选择您要购买的商品：")
    if 选择.isdigit():
            选择=int(选择)
            if 0<=选择<len(供选购的产品):
                已选产品=供选购的产品[选择]
                总价+=已选产品[1]
                购物清单.append(已选产品)
                print("------------购物清单-------------")
                for 序号,项目 in enumerate(购物清单):
                    print(序号+1,项目)
                print("您选择的“%s”已加入购物车，当前的总价=%s 元" % (已选产品[0],总价))    #这里%s的用法很特殊，表示引号中的百分号 由后面的%的内容对应替换
                print()
                input('敲回车键继续购买')
                
            else:
                print("抱歉，您选择的商品不存在！")
    elif 选择=='q':
           print("------------本次总购物清单如下-------------")
           for 序号,项目 in enumerate(购物清单):
               print(序号+1,项目)
               print("当前的总价为%s元" % 总价)
           while 1:
                   预收=input('请输入预收')
                   if 预收.isdigit():
                       预收=int(预收)
                       if 预收>=总价:
                           找零=预收-总价
                           print("预收顾客%s元，找零顾客%s元"%(预收,找零))
                           input("敲回车键退出程序")
                       if 预收<总价:
                           还需支付=总价-预收
                           print("预收顾客%s元，顾客还需要支付%s元"%(预收,还需支付))
                           input("敲回车键退出程序")
                       break
                   else:
                       print("输入错误，请重新输入")
               
           break
    else:
            print("您的选择不合法！")
```
